create table names_concat_merged(id TEXT, name1 TEXT, name2 TEXT)

insert into names_concat_merged
select min(id),
  max(case when seq = 1 then name end),
  max(case when seq = 2 then name end)
from
(
  select id, name,
    row_number() over(partition by id order by id) seq
  from names_concat
) d
group by id;
